frozen_instacart = instacart %>% 
  filter(department == 'frozen') %>% 
  add_count(product_name) %>% 
  filter(n >= 500)

frozen_instacart %>% 
  count(product_name) %>%
  mutate(product_name = as.character(product_name)) %>%
  mutate(product_name = fct_reorder(product_name, n)) %>%
    plot_ly(
      x = ~product_name, y = ~n, color = ~product_name, type = "bar", colors = "viridis") %>%
  layout(title = 'What Products (Department) Are New Yorkers Ordering?')
instacart %>% 
  count(department) %>%
  mutate(n = n/sum(n) * 100) %>%
  rename(percent = n) %>%
  mutate(department = as.character(department)) %>%
  mutate(department = fct_reorder(department, percent)) %>%
    plot_ly(
      x = ~department, y = ~percent, color = ~department, type = "bar", colors = "viridis") %>%
  layout(title = 'What Products (Department) Are New Yorkers Ordering?')
